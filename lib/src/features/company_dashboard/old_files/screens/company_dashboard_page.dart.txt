// Helper class for mapping UI cards programmatically
class _DashItem {
  final String title;
  final IconData icon;
  final Color color;
  final VoidCallback onTap;
  final int? badge;

  _DashItem({required this.title, required this.icon, required this.color, required this.onTap, this.badge});
}

class CompanyDashboardPage extends ConsumerStatefulWidget {
  const CompanyDashboardPage({super.key});

  @override
  ConsumerState<CompanyDashboardPage> createState() => _CompanyDashboardPageState();
}

class _CompanyDashboardPageState extends ConsumerState<CompanyDashboardPage> {
  void changePage(int targetIndex, String targetRouteName) {
    ref.read(companyDashboardControllerProvider.notifier).changePage(targetIndex, targetRouteName, isSubscriptionActive: true);
  }

  @override
  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context)!;
    final authState = ref.watch(authProvider);
    final company = authState.isCompanyMember ? authState.company : null;
    final isDark = Theme.of(context).brightness == Brightness.dark;

    final dashboardAsync = ref.watch(dashboardDataProvider);

    if (company == null) {
      return Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Icon(Icons.business_outlined, size: 64.sp, color: Colors.grey),
            SizedBox(height: 16.h),
            Text(
              l10n.no_company_found,
              style: TextStyle(fontSize: 18.sp, fontWeight: FontWeight.bold),
            ),
            SizedBox(height: 8.h),
            Text(l10n.not_linked_company, style: TextStyle(fontSize: 14.sp)),
          ],
        ),
      );
    }

    if (company.status == 'pending') {
      return Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Container(
              padding: EdgeInsets.all(20.r),
              decoration: BoxDecoration(color: Colors.orange.withValues(alpha: 0.1), shape: BoxShape.circle),
              child: Icon(Iconsax.timer_1_bold, size: 60.sp, color: Colors.orange),
            ),
            SizedBox(height: 24.h),
            Text(
              l10n.verification_pending,
              style: TextStyle(fontSize: 22.sp, fontWeight: FontWeight.bold),
            ),
            SizedBox(height: 12.h),
            Padding(
              padding: EdgeInsets.symmetric(horizontal: 40.w),
              child: Text(
                l10n.verification_pending_msg,
                textAlign: TextAlign.center,
                style: TextStyle(fontSize: 14.sp, color: Colors.grey[600]),
              ),
            ),
            SizedBox(height: 32.h),
            ElevatedButton(onPressed: () => context.pop(), child: Text(l10n.go_back)),
          ],
        ),
      );
    }

    return dashboardAsync.when(
      loading: () => Center(child: LoadingWidgets(size: 50.r)),
      error: (error, stackTrace) {
        dPrint(error, stackTrace: stackTrace, tag: 'DashboardError');
        return Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Icon(Icons.error_outline, size: 48.sp, color: Colors.redAccent),
              SizedBox(height: 16.h),
              Text(
                'Failed to load dashboard data',
                style: TextStyle(fontSize: 18.sp, fontWeight: FontWeight.bold),
              ),
              SizedBox(height: 8.h),
              Padding(
                padding: EdgeInsets.symmetric(horizontal: 32.w),
                child: Text(
                  error.toString(),
                  textAlign: TextAlign.center,
                  style: TextStyle(color: Colors.grey, fontSize: 13.sp),
                ),
              ),
            ],
          ),
        );
      },
      data: (dashboard) {
        final Map<String, dynamic> stats = {'inventory_value': 45000.00, 'pending_orders': dashboard.deliveryOptions};
        const String currencySymbol = "\$";

        // Mapped UI Items Based on App Features
        final manageBusinessItems = <_DashItem>[
          if (isEnabled(ref, 'company_orders', defaultValue: true))
            _DashItem(title: l10n.orders, icon: FontAwesomeIcons.clipboardList, color: Colors.deepOrange, onTap: () => changePage(6, 'orders')),
          if (isEnabled(ref, 'company_pos', defaultValue: true))
            _DashItem(title: l10n.pos, icon: FontAwesomeIcons.cashRegister, color: Colors.purple, onTap: () => changePage(5, 'pos')),
          if (isEnabled(ref, 'company_invoices', defaultValue: true))
            _DashItem(title: l10n.invoices, icon: FontAwesomeIcons.fileInvoiceDollar, color: Colors.teal, onTap: () => changePage(7, 'invoices')),
          if (isEnabled(ref, 'company_inventory', defaultValue: true))
            _DashItem(
              title: l10n.inventory,
              icon: FontAwesomeIcons.boxesStacked,
              color: Colors.green,
              onTap: () => changePage(4, 'inventory'),
              badge: dashboard.products,
            ),
          if (isEnabled(ref, 'company_offers', defaultValue: true))
            _DashItem(title: l10n.offers, icon: FontAwesomeIcons.bullhorn, color: Colors.orange, onTap: () => changePage(3, 'offers'), badge: dashboard.offers),
          if (isEnabled(ref, 'company_accounting', defaultValue: true))
            _DashItem(title: l10n.accounting, icon: FontAwesomeIcons.calculator, color: Colors.deepPurple, onTap: () => changePage(8, 'accounting')),
        ];

        final peopleItems = <_DashItem>[
          if (isEnabled(ref, 'company_members', defaultValue: true))
            _DashItem(title: l10n.members, icon: FontAwesomeIcons.users, color: Colors.brown, badge: dashboard.members, onTap: () => changePage(10, 'members')),
          if (isEnabled(ref, 'company_customers', defaultValue: true))
            _DashItem(
              title: l10n.customers,
              icon: FontAwesomeIcons.userGroup,
              color: Colors.indigoAccent,
              badge: dashboard.customers,
              onTap: () => changePage(12, 'customers'),
            ),
          if (isEnabled(ref, 'company_suppliers', defaultValue: true))
            _DashItem(title: l10n.suppliers, icon: Iconsax.shop_bold, color: Colors.purpleAccent, onTap: () => changePage(13, 'suppliers')),
          if (isEnabled(ref, 'company_my_purchases', defaultValue: true))
            _DashItem(
              title: l10n.my_purchases,
              icon: FontAwesomeIcons.bagShopping,
              color: Colors.pink,
              badge: dashboard.myPurchases,
              onTap: () => changePage(14, 'my_purchases'),
            ),
        ];

        final toolsItems = <_DashItem>[
          if (isEnabled(ref, 'company_analytics', defaultValue: true))
            _DashItem(title: l10n.analytics, icon: FontAwesomeIcons.chartLine, color: Colors.indigo, onTap: () => changePage(9, 'analytics')),
          if (isEnabled(ref, 'company_systems', defaultValue: true))
            _DashItem(
              title: l10n.systems,
              icon: FontAwesomeIcons.solarPanel,
              color: Colors.blue,
              badge: dashboard.systems,
              onTap: () => changePage(11, 'systems'),
            ),
          if (isEnabled(ref, 'company_delivery', defaultValue: true))
            _DashItem(
              title: l10n.delivery,
              icon: Icons.local_shipping,
              color: Colors.cyan,
              badge: dashboard.deliveryOptions,
              onTap: () => changePage(15, 'delivery'),
            ),
          if (isEnabled(ref, 'company_contacts', defaultValue: true))
            _DashItem(
              title: 'Contacts',
              icon: Iconsax.call_calling_bold,
              color: Colors.cyan,
              badge: dashboard.contacts,
              onTap: () => changePage(16, 'contacts'),
            ),
        ];

        final settingsItems = <_DashItem>[
          _DashItem(title: l10n.company_profile, icon: Iconsax.building_bold, color: Colors.blueGrey, onTap: () => changePage(1, 'profile')),
          _DashItem(title: l10n.subscription, icon: FontAwesomeIcons.star, color: Colors.amber, onTap: () => changePage(2, 'subscription')),
          _DashItem(title: l10n.settings, icon: Iconsax.setting_2_bold, color: Colors.blueGrey, onTap: () => context.push('/settings')),
        ];

        return SingleChildScrollView(
          padding: EdgeInsets.symmetric(horizontal: 20.w, vertical: 24.h),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              // Welcome / Header Card
              _buildModernCompanyHeader(context, company, l10n, currencySymbol, stats, isDark),
              SizedBox(height: 32.h),

              _buildDashboardSection(context, l10n.manage_business, manageBusinessItems),
              _buildDashboardSection(context, l10n.people, peopleItems),
              _buildDashboardSection(context, l10n.tools, toolsItems),
              _buildDashboardSection(context, l10n.settings, settingsItems),

              SizedBox(height: 60.h),
            ],
          ),
        );
      },
    );
  }

  Widget _buildModernCompanyHeader(
    BuildContext context,
    Company company,
    AppLocalizations l10n,
    String currencySymbol,
    Map<String, dynamic> stats,
    bool isDark,
  ) {
    return Container(
      width: double.infinity,
      padding: EdgeInsets.all(24.r),
      decoration: BoxDecoration(
        gradient: LinearGradient(
          begin: Alignment.topLeft,
          end: Alignment.bottomRight,
          colors: [
            Theme.of(context).primaryColor,
            Theme.of(context).primaryColor.withValues(alpha: isDark ? 0.6 : 0.8),
          ],
        ),
        borderRadius: BorderRadius.circular(24.r),
        boxShadow: [
          BoxShadow(
            color: Theme.of(context).primaryColor.withValues(alpha: isDark ? 0.1 : 0.3),
            blurRadius: 20,
            offset: const Offset(0, 8),
          ),
        ],
      ),
      child: Column(
        children: [
          Row(
            children: [
              Container(
                padding: EdgeInsets.all(company.logo != null ? 3.r : 16.r),
                decoration: BoxDecoration(color: Colors.white.withValues(alpha: 0.2), shape: BoxShape.circle),
                child: company.logo != null
                    ? WdImagePreview(imageUrl: company.logo!, size: 60, shape: BoxShape.circle)
                    : Text(
                        company.name.isNotEmpty ? company.name[0] : 'C',
                        style: TextStyle(fontSize: 28.sp, color: Colors.white, fontWeight: FontWeight.bold),
                      ),
              ),
              SizedBox(width: 16.w),
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text(
                      company.name,
                      maxLines: 2,
                      overflow: TextOverflow.ellipsis,
                      style: TextStyle(color: Colors.white, fontSize: 20.sp, fontWeight: FontWeight.bold, height: 1.2),
                    ),
                    SizedBox(height: 8.h),
                    Container(
                      padding: EdgeInsets.symmetric(horizontal: 10.w, vertical: 4.h),
                      decoration: BoxDecoration(
                        color: Colors.black.withValues(alpha: 0.15),
                        borderRadius: BorderRadius.circular(20.r),
                        border: Border.all(color: Colors.white.withValues(alpha: 0.2)),
                      ),
                      child: Text(
                        _getTierTranslated(context, company.tier ?? 'Standard'),
                        style: TextStyle(color: Colors.white, fontSize: 11.sp, fontWeight: FontWeight.w600, letterSpacing: 0.5),
                      ),
                    ),
                  ],
                ),
              ),
            ],
          ),
          SizedBox(height: 24.h),
          Container(
            width: double.infinity,
            padding: EdgeInsets.symmetric(horizontal: 20.w, vertical: 16.h),
            decoration: BoxDecoration(
              color: Colors.white.withValues(alpha: 0.1),
              borderRadius: BorderRadius.circular(16.r),
              border: Border.all(color: Colors.white.withValues(alpha: 0.15)),
            ),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: [
                // Cash Balance
                Expanded(
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Row(
                        children: [
                          Icon(Iconsax.wallet_3_bold, color: Colors.white.withValues(alpha: 0.8), size: 16.sp),
                          SizedBox(width: 8.w),
                          Text(
                            l10n.balance,
                            style: TextStyle(color: Colors.white.withValues(alpha: 0.9), fontSize: 13.sp),
                          ),
                        ],
                      ),
                      SizedBox(height: 6.h),
                      FittedBox(
                        fit: BoxFit.scaleDown,
                        alignment: Alignment.centerLeft,
                        child: Text(
                          NumberFormat.currency(symbol: currencySymbol).format(0.0),
                          style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold, fontSize: 18.sp),
                        ),
                      ),
                    ],
                  ),
                ),

                Container(
                  width: 1.w,
                  height: 40.h,
                  color: Colors.white.withValues(alpha: 0.3),
                  margin: EdgeInsets.symmetric(horizontal: 16.w),
                ),

                // Inventory Value
                Expanded(
                  child: Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Row(
                        children: [
                          Icon(Iconsax.box_bold, color: Colors.white.withValues(alpha: 0.8), size: 16.sp),
                          SizedBox(width: 8.w),
                          Text(
                            l10n.stock_value,
                            style: TextStyle(color: Colors.white.withValues(alpha: 0.9), fontSize: 13.sp),
                          ),
                        ],
                      ),
                      SizedBox(height: 6.h),
                      FittedBox(
                        fit: BoxFit.scaleDown,
                        alignment: Alignment.centerLeft,
                        child: Text(
                          NumberFormat.currency(symbol: currencySymbol).format(stats['inventory_value'] ?? 0.0),
                          style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold, fontSize: 18.sp),
                        ),
                      ),
                    ],
                  ),
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }

  String _getTierTranslated(BuildContext context, String tierName) {
    return tierName;
  }

  Widget _buildDashboardSection(BuildContext context, String title, List<_DashItem> items) {
    if (items.isEmpty) return const SizedBox.shrink();

    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Padding(
          padding: EdgeInsets.only(left: 4.w, top: 12.h, bottom: 12.h),
          child: Text(
            title,
            style: TextStyle(fontSize: 18.sp, fontWeight: FontWeight.bold),
          ),
        ),
        LayoutBuilder(
          builder: (context, constraints) {
            int crossAxisCount = 2;
            double childAspectRatio = 1.0;

            // Fluid responsiveness depending on max constraints
            if (constraints.maxWidth > 1400) {
              crossAxisCount = 6;
              childAspectRatio = 1.2;
            } else if (constraints.maxWidth > 1100) {
              crossAxisCount = 5;
              childAspectRatio = 1.1;
            } else if (constraints.maxWidth > 800) {
              crossAxisCount = 4;
              childAspectRatio = 1.0;
            } else if (constraints.maxWidth > 600) {
              crossAxisCount = 3;
              childAspectRatio = 1.0;
            } else {
              crossAxisCount = 2;
              childAspectRatio = 1.1; // Balanced mobile aspect ratio
            }

            return GridView.builder(
              shrinkWrap: true,
              physics: const NeverScrollableScrollPhysics(),
              gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
                crossAxisCount: crossAxisCount,
                crossAxisSpacing: 12.w,
                mainAxisSpacing: 12.h,
                childAspectRatio: childAspectRatio,
              ),
              itemCount: items.length,
              itemBuilder: (context, index) {
                final item = items[index];
                return DashboardMenuCard(title: item.title, icon: item.icon, color: item.color, badge: item.badge, onTap: item.onTap);
              },
            );
          },
        ),
        SizedBox(height: 16.h),
      ],
    );
  }
}
